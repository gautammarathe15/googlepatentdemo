// Create execution statistics

{
    TotalScenarios = 6,
    PassedScenarios = 5,
    FailedScenarios = 1,
    TotalExecutionTime = 145,
    PeakMemoryUsage = 450
};

{
    ScenarioName = "Launch Chrome",
    Status = "PASS",
    Duration = 5.2,
    Browser = "Chrome"
});

// Generate all reports

{ 
    ReportDirectory = "reports",
    ProjectName = "Google Patents Automation"
};
